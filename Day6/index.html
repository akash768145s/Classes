// App.js
function App() {
  const [user, setUser] = useState(null);

  const login = (userData) => {
    setUser(userData);
  };

  const logout = () => {
    setUser(null);
  };

  return (
    <DashboardPage 
      user={user} 
      login={login} 
      logout={logout} 
    />
  );
}

// DashboardPage.js
function DashboardPage({ user, login, logout }) {
  return (
    <div>
      <UserProfile 
        user={user} 
        logout={logout} 
      />
      <AdminPanel 
        user={user} 
        login={login} 
      />
    </div>
  );
}

// UserProfile.js
function UserProfile({ user, logout }) {
  if (!user) return <LoginForm />;

  return (
    <div>
      <h2>Welcome, {user.name}</h2>
      <button onClick={logout}>Logout</button>
    </div>
  );
}

// AdminPanel.js
function AdminPanel({ user, login }) {
  // Must receive user prop to check permissions
  if (user?.role !== 'admin') {
    return <div>Access Denied</div>;
  }

  return (
    <div>
      <h3>Admin Dashboard</h3>
      <UserManagement />
    </div>
  );
}

Cons of Props Drilling

  • Complex prop passing
  • Tightly coupled components
  • Harder to maintain
  • Performance overhead

Context API üåê

Centralized Authentication Management

// AuthContext.js
const AuthContext = createContext();

function AuthProvider({ children }) {
  const [user, setUser] = useState(null);

  const login = (userData) => {
    setUser(userData);
  };

  const logout = () => {
    setUser(null);
  };

  const hasPermission = (requiredRole) => {
    return user && user.role === requiredRole;
  };

  return (
    <AuthContext.Provider 
      value={{ 
        user, 
        login, 
        logout,
        hasPermission,
        isAuthenticated: !!user 
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}

// UserProfile.js
function UserProfile() {
  const { user, logout } = useContext(AuthContext);

  if (!user) return <LoginForm />;

  return (
    <div>
      <h2>Welcome, {user.name}</h2>
      <button onClick={logout}>Logout</button>
    </div>
  );
}

// AdminPanel.js
function AdminPanel() {
  const { hasPermission } = useContext(AuthContext);

  if (!hasPermission('admin')) {
    return <div>Access Denied</div>;
  }

  return (
    <div>
      <h3>Admin Dashboard</h3>
      <UserManagement />
    </div>
  );
}

// App.js
function App() {
  return (
    <AuthProvider>
      <Router>
        <UserProfile />
        <AdminPanel />
      </Router>
    </AuthProvider>
  );
}

Pros of Context API

  • Centralized state management
  • Simplified component logic
  • Easy permission checks
  • Global access to auth methods
