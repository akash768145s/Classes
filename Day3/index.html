What is Event Handling? ðŸ¤”

Event handling is how JavaScript listens and responds to user actions or browser events. It's the foundation of interactive web applications.

Key Concepts:

  • Event Listener: A function that "listens" for specific events
  • Event Object: Contains information about the event that occurred
  • Event Types: click, keydown, submit, mouseover, etc.
  • Event Propagation: How events move through the DOM tree

class AudioPlayer {
    constructor() {
        this.audio = new Audio();
        this.playlist = new DoublyLinkedList(); // For prev/next
        this.setupEventListeners();
    }

    setupEventListeners() {
        // Keyboard shortcuts
        document.addEventListener('keydown', (e) => {
            if (e.code === 'Space') this.togglePlay();
            if (e.code === 'ArrowLeft') this.previous();
            if (e.code === 'ArrowRight') this.next();
        });

        // Progress bar click
        this.progressBar.addEventListener('click', (e) => {
            const clickPosition = 
                (e.clientX - this.progressBar.offsetLeft) / 
                this.progressBar.offsetWidth;
            this.audio.currentTime = 
                clickPosition * this.audio.duration;
        });
    }
}

class DocumentEditor {
    constructor() {
        this.content = new Array(); // Store text chunks
        this.setupEventListeners();
    }

    setupEventListeners() {
        this.editor.addEventListener('input', (e) => {
            const change = {
                type: e.inputType,
                data: e.data,
                position: this.getCursorPosition()
            };
            this.broadcastChange(change);
        });

        this.editor.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'z') {
                e.preventDefault();
                this.undo();
            }
        });
    }
}

class DragDropManager {
    constructor() {
        this.draggingElement = null;
        this.setupEventListeners();
    }

    setupEventListeners() {
        document.addEventListener('mousedown', (e) => {
            if (e.target.matches('.card')) {
                this.draggingElement = e.target;
                this.draggingElement.classList.add('dragging');
            }
        });

        document.addEventListener('mousemove', (e) => {
            if (this.draggingElement) {
                this.updatePosition(e.clientX, e.clientY);
            }
        });

        document.addEventListener('mouseup', () => {
            if (this.draggingElement) {
                this.dropElement();
            }
        });
    }
}

class Editor {
    constructor() {
        this.cursors = new Set();
        this.setupEventListeners();
    }

    setupEventListeners() {
        // Multi-cursor selection with Alt+Click
        this.editor.addEventListener('mousedown', (e) => {
            if (e.altKey) {
                this.addCursor(e.clientX, e.clientY);
            } else {
                this.clearCursors();
                this.addCursor(e.clientX, e.clientY);
            }
        });

        // Keyboard handling for all cursors
        document.addEventListener('keydown', (e) => {
            this.cursors.forEach(cursor => {
                this.handleInput(cursor, e);
            });
        });
    }
}

class ShoppingCart {
    constructor() {
        this.items = [];
        this.setupEventListeners();
    }

    setupEventListeners() {
        // Quantity buttons
        document.querySelectorAll('.quantity-btn').forEach(button => {
            button.addEventListener('click', (e) => {
                const item = e.target.closest('.cart-item');
                const productId = item.dataset.productId;
                const input = item.querySelector('.quantity-input');
                
                if (e.target.classList.contains('increase')) {
                    input.value = parseInt(input.value) + 1;
                } else if (e.target.classList.contains('decrease')) {
                    if (parseInt(input.value) > 1) {
                        input.value = parseInt(input.value) - 1;
                    }
                }
                
                this.updateItemQuantity(productId, parseInt(input.value));
            });
        });

        // Remove buttons
        document.querySelectorAll('.remove-item-btn').forEach(button => {
            button.addEventListener('click', (e) => {
                const item = e.target.closest('.cart-item');
                const productId = item.dataset.productId;
                this.removeItem(productId);
                item.remove();
                this.updateCartSummary();
            });
        });

        // Checkout button
        document.querySelector('.checkout-btn').addEventListener('click', () => {
            this.proceedToCheckout();
        });
    }

    updateItemQuantity(productId, quantity) {
        // Update item quantity in the cart
        const itemIndex = this.items.findIndex(item => item.id === productId);
        if (itemIndex !== -1) {
            this.items[itemIndex].quantity = quantity;
            this.updateCartSummary();
        }
    }

    removeItem(productId) {
        // Remove item from the cart
        this.items = this.items.filter(item => item.id !== productId);
    }

    updateCartSummary() {
        // Recalculate and update cart totals
        const subtotal = this.items.reduce((total, item) => {
            return total + (item.price * item.quantity);
        }, 0);
        
        const shipping = subtotal > 0 ? 5.99 : 0;
        const total = subtotal + shipping;
        
        // Update the DOM
        document.querySelector('.item-count').textContent = 
            `${this.getTotalItems()} items`;
        document.querySelector('.summary-row:nth-child(1) span:last-child').textContent = 
            `$${subtotal.toFixed(2)}`;
        document.querySelector('.summary-row:nth-child(2) span:last-child').textContent = 
            `$${shipping.toFixed(2)}`;
        document.querySelector('.total span:last-child').textContent = 
            `$${total.toFixed(2)}`;
    }

    getTotalItems() {
        return this.items.reduce((count, item) => count + item.quantity, 0);
    }

    proceedToCheckout() {
        // Handle checkout process
        window.location.href = '/checkout';
    }
}
