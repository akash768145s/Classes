Step 4: Conditionals

If-Else Statement
// If-else statement function checkAge(age) { if (age >= 18) { return 'You are an adult'; } else if (age >= 13) { return 'You are a teenager'; } else { return 'You are a child'; } }
Switch Statement
// Switch statement function getDayType(day) { switch(day) { case 'Monday': case 'Tuesday': case 'Wednesday': case 'Thursday': case 'Friday': return 'Weekday'; case 'Saturday': case 'Sunday': return 'Weekend'; default: return 'Invalid day'; } }

Step 5: Loops

For Loop Example
// For loop - counts from 1 to 5 function countToFive() { let result = ''; for (let i = 1; i <= 5; i++) { result += `Count: ${i} `; } return result; }
While Loop Example
// While loop - doubles a number until it's over 100 function doubleUntil100(num) { let result = []; while (num <= 100) { result.push(num); num = num * 2; } return result; }

Step 6: Arrays and Objects

Array Example
// Array declaration let fruits = [ 'apple', 'banana', 'orange', 'grape' ]; // Array processing function processFruits(fruitArray) { let result = `We have ${fruitArray.length} fruits: `; fruitArray.forEach((fruit, index) => { result += `${index + 1}. ${fruit} `; }); return result; }
Object Example
// Object declaration let person = { name: 'Alice', age: 30, city: 'New York', skills: [ 'JavaScript', 'HTML', 'CSS' ] }; // Object processing function displayPersonInfo(personObj) { return ` Name: ${personObj.name}, Age: ${personObj.age}, City: ${personObj.city}, Skills: ${personObj.skills.join(', ')} `; }
